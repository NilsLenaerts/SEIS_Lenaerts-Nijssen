void Bytestream::readFile(std::string path)
{
    buffer.clear();
    currentByteIndex = 0;
    //read file in binary format
    std::ifstream stream(path, std::ifstream::binary);
    if (stream.is_open()) {
        uint8_t byte;
        while (stream.good()) {
            byte = stream.get();
            buffer.push_back(byte);
        }
        stream.close();
    }
    else {
        //close program if didn't read a file
        exit(EXIT_FAILURE);
    }
}
